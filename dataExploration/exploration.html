#for the unrelated class
df_nan_no = df[df.Stance == 'unrelated'].copy()
print('Number of entries in this class {0}'.format(df_nan_no.shape[0]))
df_nan_no.Headline.fillna("", inplace=True)
df_nan_no.articleBody.fillna("",inplace= True)

all_text = df_nan_no.Headline.str.cat(df_nan_no.articleBody, sep=' ')
words = nltk.word_tokenize(all_text.str.cat(sep=' ').decode('utf-8'))


stop = stopwords.words('english') 
cleanwords = [i for i in words if i not in stop and i.isalpha() and len(i) > 2]
wordcloud2 = WordCloud(
                          stopwords=STOPWORDS,
                          background_color='white',
                          width=1200,
                          height=1000
                         ).generate(" ".join(cleanwords))


plt.imshow(wordcloud2)
plt.axis('off')
plt.show()

def findAverageLen(df):
    headLst = []
    artiLst = []
    for i in range((df.shape[0])):
        try:
            headLst.append(len(df.iloc[i].Headline.split()))
            artiLst.append(len(df.iloc[i].articleBody.split()))
        except:
            print i
    return headLst,artiLst
headLst,artiLst = findAverageLen(df)
print 'The average Headline length is {0}, and averge article length is {1}'.format(np.mean(headLst),np.mean(artiLst))
print 'The max Headline length is {0}, and max article length is {1}'.format(np.max(headLst),np.max(artiLst))
print 'The min Headline length is {0}, and min article length is {1}'.format(np.min(headLst),np.min(artiLst))
print "The median Headline length is {0}, and median article length is {1}".format(np.median(headLst),np.median(artiLst))
